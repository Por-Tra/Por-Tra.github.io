  <script>
    document.querySelector('.burger').onclick = () =>
      document.querySelector('nav ul').classList.toggle('active');

    // Slider BG auto pour chaque section hero
    document.querySelectorAll('.hero').forEach(hero => {
      const slides = hero.querySelectorAll('.bg-slide');
      let idx = 0;
      setInterval(() => {
        slides[idx].classList.remove('active');
        idx = (idx + 1) % slides.length;
        slides[idx].classList.add('active');
      }, 4000);
    });

    // Intersection Observer simple pour les animations
    const observerOptions = {
      threshold: 0.2,
      rootMargin: '0px 0px -100px 0px'
    };

    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          entry.target.classList.add('animate');
          
          // Animer les barres de progression simplement
          const stackItems = entry.target.querySelectorAll('.stack-item');
          stackItems.forEach((item, index) => {
            setTimeout(() => {
              const progress = item.getAttribute('data-progress');
              const fill = item.querySelector('.progress-fill');
              if (fill) {
                fill.style.setProperty('--progress-width', `${progress}%`);
                fill.style.width = `${progress}%`;
              }
            }, 100 + (index * 50));
          });
        }
      });
    }, observerOptions);

    // Observer la section stack
    const stackSection = document.querySelector('.stack');
    if (stackSection) {
      observer.observe(stackSection);
    }

    // Effets hover simples — only attach on devices that support hover to avoid touch interference
    if (window.matchMedia('(hover: hover) and (pointer: fine)').matches) {
      document.querySelectorAll('.stack-item-box').forEach(box => {
        box.addEventListener('mouseenter', function() {
          this.classList.add('is-hover');
        });
        box.addEventListener('mouseleave', function() {
          this.classList.remove('is-hover');
        });
      });
    }

  </script>
